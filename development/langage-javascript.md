### Un exemple de code commenté ✔️

```javascript
function generateStairs(row) {
    let output = '';
    let result = [];
    for (let i = 1; i <= row; i++) {
        for (let j = 1; j <= i; j++) {
            output += '*' + '  ';
        }
        result.push(output);
        output = '';
    }
    return result;
}

generateStairs(5);
```
